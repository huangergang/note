线性表的数组存储

```c
#define _CRT_SECURE_NO_WARNINGS
#define MAXSIZE 100
#include<stdio.h>
#include<stdlib.h>
typedef int ElemType;
typedef struct SqList
{
	ElemType Data[MAXSIZE];
	int length;
};

// 我也不清楚为什吗这样初始化......fuck!!!
SqList* Init_List()
{
	SqList *L = (SqList*)malloc(sizeof(SqList));
	L->length = 0;
	return L;
}

// 返回表的长度length
int Length(SqList L)
{
	return L.length;
}

// 增加n个表中数据
void Add_List(SqList*L)
{
	int n;
	printf("输入添加个数：\n");
	scanf("%d", &n);
	n = L->length + n;
	for (int i = L->length; i <n ; i++)
	{
		scanf("%d",&L->Data[i]);
		L->length++;
	}
	printf("成功添加%d个元素到线性表！\n", L->length);
}

// 显示表中数据函数
void Show_List(SqList L)
{
	printf("表中数据如下:\n");
	for (int i = 0; i < L.length; i++)
	{
		printf("%d ", L.Data[i]);
	}
	printf("\n");
}

// 查找元素位置
void Locate_Elem(SqList L)
{
	printf("输入查找元素\n");
	int e;
	scanf("%d", &e);
	if (L.length == 0)
		printf("表空，你查个屁！\n");
	else 
	{
		int i;
		for ( i = 0; i < L.length; i++) {
			if (L.Data[i] == e) {
				printf("您要的数据在第%d位\n", ++i);
				break;
			}
		}
		if (i == L.length)
		{
			printf("没有该元素!\n");
		}
	}
}

// 按位置查找
void Locate_position(SqList L)
{
	printf("输入查找位置\n");
	int e;
	scanf("%d", &e);
	if (L.length == 0)
		printf("表空，你查个屁！\n");
	else if (e<1 || e>L.length)
		printf("查找位置不合法！\n");
	else
	printf("该位置的元素为：%d\n", L.Data[e - 1]);
}

// 插入元素
void insert_List(SqList* L)
{
	int Elem;
	printf("输入插入元素:\n");
	scanf("%d", &Elem);
	int position;
	printf("输入插入位置:\n");
	scanf("%d", &position);
	 if (position >L->length || position <= 0)
	{
		printf("插入位置不合法！\n");
	} 
	 else {

		 L->length++;

		 for (int i = L->length; i >= position; i--)
		 {
			 L->Data[i] = L->Data[i - 1];
		 }
		 L->Data[position - 1] = Elem;
		 printf("插入成功！\n");
	 }
}

// 删除元素
void Delete_List(SqList*L)
{
	int select;
	printf("按位置删除（1）or 按数据删除（2）\n");
	scanf("%d", &select);
	if (select == 1)
	{
		int e;
		printf("输入删除位置：\n");
		scanf("%d", &e);
		if (e <= 0 || e > L->length) {
			printf("输入位置不和法！\n");
		}
		else {
			for (int i = e - 1; i <= L->length; i++)
			{
				L->Data[i] = L->Data[i + 1];
			}
			L->Data[L->length] = 0;
			L->length--;
			printf("删除成功！\n");
		}
	}
	else {
		int e;
		printf("输入删除元素：\n");
		scanf("%d", &e);
		int count= 0;
		while (L->Data[count] != e) {
			count++;
			// 判断循环完也没找到e元素
			if (L->Data[count] == e) {
				for (int i = count; i < L->length; i++)
				{
					L->Data[i] = L->Data[i + 1];
				}
				L->Data[L->length] = 0;
				L->length--;
				printf("删除成功！\n");
				break;
			}
			else {
				printf("该表中没有元素： %d\n",e);
				break;
			}
		}
	}
	
}

// enum
void List_enum()
{
	printf("\t-------------线性表的数组存储-------------\n");
	printf("\t\t1.添加元素\n");
	printf("\t\t2.显示表中数据\n");
	printf("\t\t3.按数值查找数据位置\n");
	printf("\t\t4.按位置查找表中数据\n"); 
	printf("\t\t5.插入元素\n");
	printf("\t\t6.删除元素\n");
	printf("\t\t7.查询表长\n");
	printf("\t\t8.清屏\n");
	printf("\t------------------------------------------\n");

}

// 主函数
int main()
{ 
	SqList* List;
	List=Init_List();   
	int Options;
	List_enum();
	do
	{
		printf("选择您的指令,(输入-1退出)\n");
		scanf("%d", &Options);
		switch(Options)
		{
		case 1:Add_List(List); break;
		case 2:Show_List(*List); break;
		case 3:Locate_Elem(*List); break;
		case 4:Locate_position(*List); break;
		case 5:insert_List(List); break;
		case 6:Delete_List(List); break;
		case 7:printf("该线性表长为：%d\n", Length(*List)); break;
		case 8:system("cls"); List_enum(); break;
		case -1:exit(0); break;
		}
	} while (true);

	return 0;
}
```
